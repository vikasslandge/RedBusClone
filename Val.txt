function validateLogin(){
    var mob = document.getElementById("MobileNumber").value;
    var pass = document.getElementById("Password").value;

    var mob_pattern= /^[7-9]{1}[0-9]{9}/; //start with 7/8/9 and remaining 9 will 
    var pass_pattern=/^((?=.*\d)(?=.*[Aa-Zz])).{4,10}/; //number 0-9 with 1 char between 4-10 digits

    if (mob == "") {
        document.getElementById("mob_error").innerHTML = "*This field is required";
        return false;
    }
    if (mob != "") {
        document.getElementById("mob_error").innerHTML = "";

    }
    if (pass == "") {
        document.getElementById("password_error").innerHTML = "*This field is required";
        return false;
    }
    if (pass != "") {
        document.getElementById("password_error").innerHTML = "";

    }

    if(!mob_pattern.test(mob)){
        document.getElementById("mob_error").innerHTML="*Mobile number is not correct";
        return false;
    }

    if(!pass_pattern.test(mob)){
        document.getElementById("password_error").innerHTML="*Password not in correct format";
        return false;
    }
    

    else {
        return true;
    }
}
